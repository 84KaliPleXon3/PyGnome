    """
    Create output directory for rendered images.

    """

        images_dir = os.path.join(os.getcwd(), 'images')

    print 'images_dir is:', images_dir

        print 'removing...', images_dir
        shutil.rmtree(images_dir)
    os.mkdir(images_dir)
